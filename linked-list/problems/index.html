      #include <iostream>
#include <forward_list>

using namespace std;

int main() {
    forward_list<int> list;
    for (int i = 0; i < 10; i++)
        list.push_front(i);
    for (int i = 0; i < 5; i++)
        list.pop_front();
    cout << endl;
    for (int i : list)
        cout << i << " ";
}
